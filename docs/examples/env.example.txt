# Mini Agent Python 环境变量示例
#
# 使用方式：复制此文件为 .env 或在终端中设置环境变量
# 注意：敏感凭据请放在 config.user.json 的 secrets 部分

# ============================================================================
# API 凭据（必填）
# ============================================================================

# LLM API 密钥
OPENAI_API_KEY=sk-xxx-your-api-key-here

# API 基础 URL（可指向兼容 OpenAI 的第三方服务）
MINIAGENT_MODEL_BASE_URL=https://api.openai.com/v1

# 主模型名称
MINIAGENT_MODEL_MODEL=gpt-4o-mini

# Web 搜索 API 密钥（可选）
TAVILY_API_KEY=tvly-xxx-your-tavily-key-here

# ============================================================================
# 飞书配置（可选，--feishu 启动时需要）
# ============================================================================

FEISHU_APP_ID=cli_xxx-your-app-id
FEISHU_APP_SECRET=xxx-your-app-secret
FEISHU_VERIFICATION_TOKEN=xxx-your-verification-token

# ============================================================================
# Agent 执行配置（可选）
# ============================================================================

# Agent 最大对话轮次
MINIAGENT_AGENT_MAX_TURNS=400

# 分步执行时每步 ReAct 子循环上限
MINIAGENT_STEP_MAX_TURNS=48

# 单个工具超时秒数
MINIAGENT_AGENT_TOOL_TIMEOUT=60

# HTTP 请求超时秒数
MINIAGENT_AGENT_HTTP_TIMEOUT=120

# ============================================================================
# 飞书核心体验（可选）
# ============================================================================

# 回复模式：reply=回复入站消息，create=新建消息
MINIAGENT_FEISHU_REPLY_TARGET=reply

# 卡片按钮回调注册
MINIAGENT_FEISHU_CARD_ACTION_ROUTER=1

# ============================================================================
# 时区配置（可选）
# ============================================================================

# 标准时区环境变量
TZ=Asia/Shanghai

# ============================================================================
# 状态目录（可选）
# ============================================================================

# 运行时状态目录
MINIAGENT_PATHS_STATE_DIR=workspaces
